def setColor(s):
    if s == 'M':
        fill_color = 'blue'
    else:
        fill_color = 'pink'
    return fill_color

def setHtml(row): 
    html = "Arrest #: " + str(row['arrest']) + "\n" + \
    "Race: " + row['race'] + "\n" + \
    "Age: " + str(row['age']) + "\n" + \
    "Incident Offense: " + row['incidentOffense']
    return html

for col, row in arrest_table.sample(250).iterrows():

    folium.Marker(location = [row['lat'], row['long']], popup = setHtml(row),
                      icon = folium.Icon(color = setColor(row['sex']))).add_to(map_osm)

This is an interactive criminal data map in baltimore. I pick up 250 samples and map the sample using markers that indicate the sex of the offender.
